## The function is currently defined as
function (n, se) 
{
    succeeded <- TRUE
    comment <- ""
    if (!is.na(n)) {
        if (n <= 1) {
            comment <- "n<=1"
            succeeded <- FALSE
        }
    }
    if (se <= 0) {
        comment <- paste("se<=0", comment, sep = ", ")
        succeeded <- FALSE
    }
    return(list(succeeded = succeeded, comment = comment))
  }
